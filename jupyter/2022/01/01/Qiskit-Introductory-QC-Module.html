from qiskit import QuantumCircuit
qc = QuantumCircuit(3,3)
qc.measure([0,1,2],[0,1,2])
qc.draw()

from qiskit.providers.aer import AerSimulator
sim = AerSimulator()

job = sim.run(qc)
result = job.result()
